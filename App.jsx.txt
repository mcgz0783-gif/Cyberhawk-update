async function askCyberBot(messages) {
  // Detect environment: use Gemini if API key is set, else Anthropic (artifact preview)
  const geminiKey = typeof import !== "undefined" && typeof import.meta !== "undefined"
    ? import.meta?.env?.VITE_GEMINI_API_KEY
    : null;

  if (geminiKey) {
    // ── Google Gemini API ──────────────────────────────────────────
    const contents = messages.map(m => ({
      role: m.role === "assistant" ? "model" : "user",
      parts: [{ text: m.content }],
    }));
    const res = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-pro:generateContent?key=${geminiKey}`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          system_instruction: { parts: [{ text: SYSTEM_PROMPT }] },
          contents,
          generationConfig: { maxOutputTokens: 1024, temperature: 0.7 },
        }),
      }
    );
    const data = await res.json();
    return data.candidates?.[0]?.content?.parts?.[0]?.text || "I couldn't generate a response.";
  } else {
    // ── Anthropic Claude API (artifact / fallback) ─────────────────
    const res = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1024,
        system: SYSTEM_PROMPT,
        messages,
      }),
    });
    const data = await res.json();
    return data.content?.find(b => b.type === "text")?.text || "I couldn't generate a response.";
  }
}

// ─── AI CHAT WIDGET ───────────────────────────────────────────────────
function AIChatWidget() {
  const [open, setOpen] = useState(false);
  const [messages, setMessages] = useState([
    { role:"assistant", content:"👋 I'm **CyberBot** — your AI security expert. Ask me anything about cybersecurity, current threats, this platform's content, or your career in security. How can I help you today?" }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef();
  const inputRef = useRef();

  useEffect(() => { if(open && bottomRef.current) bottomRef.current.scrollIntoView({ behavior:"smooth" }); }, [messages, open]);
  useEffect(() => { if(open) setTimeout(()=>inputRef.current?.focus(), 100); }, [open]);

  const send = async () => {
    const text = input.trim();
    if (!text || loading) return;
    const userMsg = { role:"user", content:text };
    setMessages(m=>[...m, userMsg]);
    setInput("");
    setLoading(true);
    try {
      const history = [...messages, userMsg].filter(m=>m.role!=="system").map(m=>({ role:m.role, content:m.content }));
      const reply = await askCyberBot(history);
      setMessages(m=>[...m, { role:"assistant", content:reply }]);
    } catch {
      setMessages(m=>[...m, { role:"assistant", content:"⚠️ Connection error. Please try again." }]);
    }
    setLoading(false);
  };

  const handleKey = e => { if(e.key==="Enter" && !e.shiftKey) { e.preventDefault(); send(); } };

  const renderMsg = text =>
    text.split("\n").map((line, i) => (
      <span key={i}>{line.replace(/\*\*(.*?)\*\*/g, (_,m)=>m)}<br/></span>
    ));

  const SUGGESTIONS = ["What is Zero Trust?","Explain ransomware-as-a-service","How do I start a career in cybersecurity?","Latest CVE to patch?"];

  return (
    <div className="chat-widget">
      {open && (
        <div className="chat-panel">
          {/* Header */}
          <div style={{ padding:"16px 18px", borderBottom:`1px solid ${T.border}`, display:"flex", alignItems:"center", gap:12, background:T.card }}>
            <div style={{ width:36, height:36, background:`linear-gradient(135deg,${T.accent},${T.green})`, display:"flex", alignItems:"center", justifyContent:"center", fontSize:18, flexShrink:0 }}>🤖</div>
            <div style={{ flex:1 }}>
              <div style={{ fontFamily:"'Exo 2',sans-serif", fontWeight:800, fontSize:14, color:T.white }}>CyberBot AI</div>
              <div style={{ fontSize:11, color:T.green, display:"flex", alignItems:"center", gap:5 }}>
                <span style={{ width:6, height:6, background:T.green, borderRadius:"50%", display:"inline-block", animation:"blink 2s infinite" }} />
                Powered by Google Gemini
              </div>
            </div>
            <button onClick={()=>setOpen(false)} style={{ background:"none", border:"none", color:T.muted, cursor:"pointer", fontSize:18 }}>✕</button>
          </div>

          {/* Messages */}
          <div style={{ flex:1, overflowY:"auto", padding:16, display:"flex", flexDirection:"column", gap:12, minHeight:0, maxHeight:380 }}>
            {messages.map((msg, i) => (
              <div key={i} className="chat-msg" style={{ display:"flex", justifyContent:msg.role==="user"?"flex-end":"flex-start" }}>
                <div style={{
                  maxWidth:"85%", padding:"10px 14px",
                  background:msg.role==="user"?`${T.accent}22`:T.card,
                  border:`1px solid ${msg.role==="user"?T.accent:T.border}`,
                  fontSize:13, color:T.text, lineHeight:1.7,
                }}>
                  {renderMsg(msg.content)}
                </div>
              </div>
            ))}
            {loading && (
              <div style={{ display:"flex", justifyContent:"flex-start" }}>
                <div style={{ padding:"12px 16px", background:T.card, border:`1px solid ${T.border}` }}>
                  <span className="typing-dot"/><span className="typing-dot"/><span className="typing-dot"/>
                </div>
              </div>
            )}
            <div ref={bottomRef}/>
          </div>

          {/* Suggestions (only when few messages) */}
          {messages.length <= 2 && (
            <div style={{ padding:"8px 12px", display:"flex", gap:6, flexWrap:"wrap", borderTop:`1px solid ${T.border}33` }}>
              {SUGGESTIONS.map(s=>(
                <button key={s} onClick={()=>{ setInput(s); inputRef.current?.focus(); }}
                  style={{ background:`${T.accent}12`, border:`1px solid ${T.accent}33`, color:T.accent, padding:"4px 10px", fontSize:11, fontFamily:"'Rajdhani',sans-serif", cursor:"pointer", borderRadius:2 }}>
                  {s}
                </button>
              ))}
            </div>
          )}

          {/* Input */}
          <div className="chat-input-row">
            <input ref={inputRef} className="chat-input" placeholder="Ask a security question..." value={input} onChange={e=>setInput(e.target.value)} onKeyDown={handleKey} />
            <button className="chat-send" onClick={send} disabled={loading}>
              {loading?"⏳":"➤"}
            </button>
          </div>
        </div>
      )}

      <button className="chat-fab" onClick={()=>setOpen(o=>!o)} title="Ask CyberBot AI">
        {open ? "✕" : "🤖"}
      </button>
    </div>
  );
}

              <div style={{ display:"flex", gap:10 }}>
                {[["𝕏","Twitter"],["in","LinkedIn"],["gh","GitHub"],["▶","YouTube"]].map(([icon,label])=>(
                  <div key={label} title={label} style={{ width:42, height:42, border:`1px solid ${T.border}`, display:"flex", alignItems:"center", justifyContent:"center", cursor:"pointer", color:T.muted, fontSize:13, transition:"all .2s" }}
                    onMouseEnter={e=>{e.currentTarget.style.borderColor=T.accent;e.currentTarget.style.color=T.accent;}}
                    onMouseLeave={e=>{e.currentTarget.style.borderColor=T.border;e.currentTarget.style.color=T.muted;}}>{icon}</div>
                ))}
              </div>
            </div>
          </div>
          <div style={{ background:T.card, border:`1px solid ${T.border}`, padding:36 }}>
            {sent ? (
              <div style={{ textAlign:"center", padding:"56px 20px" }}>
                <div style={{ fontSize:52, marginBottom:18 }}>✅</div>
                <h3 style={{ fontFamily:"'Exo 2',sans-serif", fontWeight:800, fontSize:22, color:T.green, marginBottom:10 }}>MESSAGE SENT</h3>
                <p style={{ color:T.muted }}>We'll get back to you within 24 hours. Stay secure.</p>
                <button className="btn-ghost" onClick={()=>setSent(false)} style={{ marginTop:22 }}>SEND ANOTHER</button>
              </div>
            ) : (
              <>
                <h3 style={{ fontFamily:"'Exo 2',sans-serif", fontWeight:800, fontSize:19, color:T.white, marginBottom:26, letterSpacing:1 }}>SEND A MESSAGE</h3>
                <div style={{ display:"grid", gridTemplateColumns:"1fr 1fr", gap:14, marginBottom:14 }}>
                  <div><label style={{ display:"block", fontSize:11, color:T.muted, fontFamily:"'Share Tech Mono',monospace", letterSpacing:1, marginBottom:7 }}>FULL NAME *</label><input className="input-f" placeholder="John Doe" value={form.name} onChange={e=>upd("name",e.target.value)}/></div>
                  <div><label style={{ display:"block", fontSize:11, color:T.muted, fontFamily:"'Share Tech Mono',monospace", letterSpacing:1, marginBottom:7 }}>EMAIL *</label><input className="input-f" placeholder="you@domain.com" value={form.email} onChange={e=>upd("email",e.target.value)}/></div>
                </div>
                <div style={{ marginBottom:14 }}><label style={{ display:"block", fontSize:11, color:T.muted, fontFamily:"'Share Tech Mono',monospace", letterSpacing:1, marginBottom:7 }}>SUBJECT</label><input className="input-f" placeholder="Topic of inquiry" value={form.subject} onChange={e=>upd("subject",e.target.value)}/></div>
                <div style={{ marginBottom:22 }}><label style={{ display:"block", fontSize:11, color:T.muted, fontFamily:"'Share Tech Mono',monospace", letterSpacing:1, marginBottom:7 }}>MESSAGE *</label><textarea className="input-f" placeholder="Your message..." value={form.message} onChange={e=>upd("message",e.target.value)}/></div>
                <button className="btn-primary" onClick={()=>{if(form.name&&form.email&&form.message)setSent(true);}} style={{ width:"100%", padding:"13px", fontSize:13 }}>SEND MESSAGE →</button>
                <p style={{ marginTop:10, fontSize:11, color:T.muted, textAlign:"center" }}>We never share your data. Privacy policy applies.</p>
              </>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}

// ─── APP ROOT ─────────────────────────────────────────────────────────
export default function App() {
  const [page, setPage] = useState("Home");
  const [modal, setModal] = useState({ item:null, type:null });

  const openRead = (item, type) => setModal({ item, type });
  const closeRead = () => setModal({ item:null, type:null });

  const navigate = p => { setPage(p); window.scrollTo(0,0); };

  const pages = {
    Home: <HomePage setPage={navigate} onRead={openRead}/>,
    News: <NewsPage onRead={openRead}/>,
    Insights: <InsightsPage onRead={openRead}/>,
    Blog: <BlogPage onRead={openRead}/>,
    Books: <BooksPage onRead={openRead}/>,
    About: <AboutPage/>,
    Contact: <ContactPage/>,
  };

  return (
    <div style={{ minHeight:"100vh", background:T.bg }}>
      <div className="noise"/>
      <Navbar page={page} setPage={navigate}/>
      <main>{pages[page] || pages.Home}</main>
      <Footer setPage={navigate}/>
      <AIChatWidget/>
      {modal.item && <ReadModal item={modal.item} type={modal.type} onClose={closeRead}/>}
    </div>
  );
}
